#include<bits/stdc++.h>
using namespace std;

  

class TS{

    private:

        int n;

        vector<int> time_;

        vector<vector<int>> prereqs;

        vector<int> state;

        vector<int> order;

        bool hasCycle;

  

        void dfs(int node){

            state[node] = 1;

  

            for(int pre : prereqs[node]){

                if(state[pre] == 0){

                    dfs(pre);

                    if(hasCycle){

                        return ;

                    }

                }

                else if(state[pre] == 1){

                    hasCycle = 1;

                    return;

                }

            }

            state[node] = 2;

            order.push_back(node);

        }

    public:

        TS(int n, vector<int>& taskTimes, vector<vector<int>>& deps) : n(n), time_(taskTimes), prereqs(n), state(n,0), hasCycle(false){

            for(auto& edge : deps){

                int u = edge[0];

                int v = edge[1];

                prereqs[v].push_back(u);

            }

        }

  

        void solve(){

            for(int i =0; i < n; i++){

                if(state[i] == 0){

                    dfs(i);

                    if(hasCycle){

                        cout << "cycle" << "\n";

                        return;

                    }

                }

            }

            for (int node : order) cout << node << " ";

            cout << "\n";

        }

};

  

int main(){

    int n = 4;

    vector<int> taskTimes = {3,2,4,5};

  

    vector<vector<int>> deps = {{0,2},{1,2},{2,3}};

  

    TS ts1(n, taskTimes, deps);

    ts1.solve();

}

#include <bits/stdc++.h>
using namespace std;

class TaskSchedulerKahn {
private:
    int n;
    vector<int> time_;
    vector<vector<int>> graph; // Forward graph: u -> dependents
    vector<int> indegree;

public:
    TaskSchedulerKahn(int n, vector<int>& taskTimes, vector<vector<int>>& deps)
        : n(n), time_(taskTimes), graph(n), indegree(n, 0) {
        
        // Build graph and indegree count
        for (const auto& edge : deps) {
            int u = edge[0]; // prerequisite
            int v = edge[1]; // dependent
            graph[u].push_back(v);
            indegree[v]++;
        }
    }

    void solveSequential() {
        queue<int> q;
        // Add all nodes with no prerequisites to the queue
        for (int i = 0; i < n; ++i) {
            if (indegree[i] == 0) {
                q.push(i);
            }
        }

        vector<int> topoOrder;
        while (!q.empty()) {
            int u = q.front();
            q.pop();
            topoOrder.push_back(u);

            // "Complete" task u, so reduce indegree of its dependents
            for (int v : graph[u]) {
                indegree[v]--;
                if (indegree[v] == 0) {
                    q.push(v);
                }
            }
        }

        if (topoOrder.size() < n) {
            cout << "Result: -1 (cycle detected)\n";
        } else {
            int totalTime = 0;
            for (int t : time_) totalTime += t;

            cout << "Execution Order: ";
            for (int node : topoOrder) cout << node << " ";
            cout << "\n";

            cout << "Total Sequential Time: " << totalTime << "\n";
        }
    }
};

int main() {
    int n = 4;
    vector<int> taskTimes = {3, 2, 4, 5};
    vector<vector<int>> deps = {{0, 2}, {1, 2}, {2, 3}};

    cout << "=== Kahn's Algorithm - No Cycle Case ===\n";
    TaskSchedulerKahn ts1(n, taskTimes, deps);
    ts1.solveSequential();

    cout << "\n=== Kahn's Algorithm - Cycle Case ===\n";
    vector<vector<int>> cyclicDeps = {{0, 1}, {1, 2}, {2, 0}};
    TaskSchedulerKahn ts2(3, taskTimes, cyclicDeps); // Using 3 nodes for this cycle
    ts2.solveSequential();

    return 0;
}
